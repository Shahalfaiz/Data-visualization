import matplotlib.pyplot as plt

x = df['area']
y = df['poptotal']
dot_size = df['dot_size']
category_mapping = {category: i for i, category in enumerate(df['category'].unique())}
colors = df['category'].map(category_mapping)
plt.scatter(x, y, s=dot_size, c=colors, cmap='plasma', alpha=0.9, edgecolors='black', linewidth=1)
plt.xlabel('Area')
plt.ylabel('Population Total')
plt.title('Bubble Chart: Area vs Population Total')
plt.show()
